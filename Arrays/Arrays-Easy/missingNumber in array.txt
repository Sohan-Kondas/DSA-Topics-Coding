static int missingNumber(int []a, int N) {
        int sum=(N*(N+1))/2;
        int s2=0;
        for(int i=0; i<N-1; i++) {
            s2+=a[i];
        }
        int missingNum=sum-s2;
        return missingNum;
}

// optimal approach 2
static int missingNumber(int []a, int N) {
        int xor1=0, xor2=0;
        for(int i=0; i<N-1; i++) {
            xor2=xor2^a[i];
            xor1=xor1^(i+1);
        }
        xor1=xor1^N;
        return (xor1^xor2);
}
